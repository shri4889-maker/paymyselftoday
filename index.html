const incomeCurrency = document.getElementById('incomeCurrency').value; const clientCountry = document.getElementById('clientCountry').value; const yourCountry = document.getElementById('country').value; // e.g., "IN" const businessType = document.getElementById('businessType').value; // Convert income to INR const rate = exchangeRates[incomeCurrency] || 1; const incomeINR = incomeForeign * rate; // Convert expenses to INR (assume expenses in same currency as income — or add toggle later) const expensesINR = expenses * rate; const netProfitINR = incomeINR - expensesINR; if (netProfitINR <= 0) { alert("No profit this month — don’t pay yourself from business funds!"); return; } let incomeTaxINR = 0; let gstINR = 0; let tdsINR = 0; // INDIA-SPECIFIC LOGIC if (yourCountry === "IN") { // GST: Applies if registered AND client is outside India (export of service) // Rule: If client is outside India → GST is 0% (but you must be registered to claim) // Simplified: If GST registered + foreign client → no GST charged, but you can claim ITC // For simplicity, we assume no GST output — but user may have paid GST on expenses // So we skip GST calculation on income — but mention in disclaimer // Income Tax (same as before) if (netProfitINR <= 300000) incomeTaxINR = 0; else if (netProfitINR <= 600000) incomeTaxINR = (netProfitINR - 300000) * 0.05; else if (netProfitINR <= 900000) incomeTaxINR = 15000 + (netProfitINR - 600000) * 0.10; else if (netProfitINR <= 1200000) incomeTaxINR = 45000 + (netProfitINR - 900000) * 0.15; else if (netProfitINR <= 1500000) incomeTaxINR = 90000 + (netProfitINR - 1200000) * 0.20; else incomeTaxINR = 150000 + (netProfitINR - 1500000) * 0.30; // TDS (if client is in India → 10% TDS may apply) // If client is foreign → usually no TDS (unless they have Indian presence) // Simplified: Only apply TDS if clientCountry === "IN" if (clientCountry === "IN") { tdsINR = incomeINR * 0.10; } // Show disclaimer for Indian users const disclaimer = document.getElementById('disclaimer'); disclaimer.innerHTML = ` 💡 For Indian Freelancers: - Income converted at 1 ${incomeCurrency} = ₹${rate.toFixed(2)}
- GST not charged on foreign clients (0% under export of services)
- TDS (10%) applied only if client is in India
- Tax calculated under New Regime (FY 2025-26). Consult CA. `; disclaimer.classList.remove('hidden'); } // FOR OTHER COUNTRIES (keep existing logic) else { const countryTaxRates = { US: { income: 0.22, social: 0.153 }, CA: { income: 0.15, social: 0.12 }, GB: { income: 0.20, social: 0.12 }, AU: { income: 0.325, social: 0 }, OTHER: { income: 0.20, social: 0.10 } }; const rates = countryTaxRates[yourCountry] || countryTaxRates.OTHER; incomeTaxINR = netProfitINR * rates.income; const socialTax = netProfitINR * rates.social; incomeTaxINR += socialTax; } const totalTaxINR = incomeTaxINR + tdsINR; const takeHomeINR = netProfitINR - totalTaxINR; // DISPLAY RESULTS (always in ₹ for Indian users) const symbol = yourCountry === "IN" ? "₹" : "$"; document.getElementById('fedTax').textContent = symbol + incomeTaxINR.toFixed(0); document.getElementById('seTax').textContent = symbol + tdsINR.toFixed(0); // Show TDS here document.getElementById('totalTax').textContent = symbol + totalTaxINR.toFixed(0); document.getElementById('netPay').textContent = symbol + Math.max(0, takeHomeINR).toFixed(0); document.getElementById('stateName').textContent = yourCountry; // Also show original currency take-home (optional) const takeHomeOriginal = takeHomeINR / rate; document.getElementById('originalCurrencyNote').innerHTML = yourCountry === "IN" ? `(~ ${incomeCurrency} ${takeHomeOriginal.toFixed(2)} in original currency)` : ""; // Show results document.getElementById('calculator').classList.add('hidden'); document.getElementById('results').classList.remove('hidden'); }
